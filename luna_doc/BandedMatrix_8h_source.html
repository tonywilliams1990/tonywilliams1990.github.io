559  template <typename T>
560  inline void BandedMatrix<T>::decompose( Matrix<T>& au, Matrix<T>& al,
561  Vector<int>& index, double& d )
562  {
563  const double TINY( 1.0e-40 );
564  std::size_t i, j, k, l, mm;
565  T dum;
566  mm = M1 + M2 + 1;
567  l = M1;
568  for ( i = 0; i < M1; i++ ) { // Rearrange the storage a bit
569  for ( j = M1 - i; j< mm ; j++ ){
570  au( i, j - l ) = au( i, j );
571  }
572  l--;
573  for ( j = mm - l - 1; j < mm; j++ ){
574  au( i, j ) = 0.0;
575  }
576  }
577  d = 1.0;
578  l = M1;
579  for ( k = 0; k < N; k++ ) {
580  dum = au( k, 0 );
581  i = k;
582  if ( l < N ){ l++; }
583  for ( j = k + 1; j < l; j++ ) { // Find the pivot element
584  if ( std::abs( au( j, 0 ) ) > std::abs( dum ) ) {
585  dum = au( j, 0 );
586  i = j;
587  }
588  }
589  index[ k ] = i + 1;
590  if ( std::abs( dum ) == 0.0 ) au( k, 0 ) = TINY;
591  // Singular matrix but proceed anyway with TINY pivot
592  if (i != k) {
593  d = -d;
594  for ( j = 0; j < mm; j++ ){
595  std::swap( au( k, j ), au( i, j ) );
596  }
597  }
598  for ( i = k + 1; i < l; i++ ) {
599  dum = au( i, 0 ) / au( k, 0 );
600  al( k, i - k - 1 ) = dum;
601  for ( j = 1; j < mm; j++ ){
602  au( i, j - 1 ) = au( i, j ) - dum * au( k, j );
603  }
604  au( i, mm - 1 ) = 0.0;
605  }
606  }
