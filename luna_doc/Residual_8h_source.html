120  template <typename T>
121  inline void Residual<T>::jacobian( const Vector<T>& state,
122  Matrix<T>& jac ) const
123  {
124  Vector<T> new_state( state );
125  // evaluation of the function
126  Vector<T> f_at_new_state( ORDER_OF_SYSTEM, 0.0 );
127  // default is to FD the Jacobian
128  for ( std::size_t i = 0; i < NUMBER_OF_VARS; ++i )
129  {
130  new_state[ i ] += DELTA;
131  residual_fn( new_state, f_at_new_state );
132  new_state[ i ] -= DELTA;
133  jac.set_col( i, ( f_at_new_state - FN_AT_LAST_STATE ) / DELTA );
134  }
135  }
