243  inline Vector< std::complex<double> > Polynomial<T>::cubic_solve( const T& a,
244  const T& b, const T& c, const T& d )
245  {
246  this->cubic( a, b, c, d );
247  Vector< std::complex<double> > result( 3 );
248 
249  T e, f, g;
250  e = b / a;
251  f = c / a;
252  g = d / a;
253 
254  T Q, R;
255  Q = ( e * e - 3. * f ) / 9.;
256  R = ( 2. * e * e * e - 9. * e * f + 27. * g ) / 54.;
257 
258  std::complex<double> A, B;
259  A = - std::pow( R + std::sqrt( R * R - Q * Q * Q ) , 1. / 3. );
260  if ( A != 0. ) {
261  B = Q / A;
262  } else {
263  B = 0.;
264  }
265 
266  result[ 0 ] = A + B - ( e / 3. );
267  std::complex<double> first, second;
268  std::complex<double> i( 0.0, 1.0 );
269  first = - 0.5 * ( A + B ) - ( e / 3. );
270  second = i * 0.5 * std::sqrt( 3. ) * ( A - B );
271  result[ 1 ] = first + second;
272  result[ 2 ] = first - second;

355  const double EPS( 1.0e-14 );
356  std::size_t i, its;
357  std::complex<double> x, b, c;
358 
359  Vector< std::complex<double> > a( N + 1 );
360  for ( std::size_t n = 0; n < N + 1; n++ )
361  {
362  a[ n ] = COEFFS[ n ];
363  }
364 
365  Vector< std::complex<double> > ad( a );
366 
367  for ( int j = N - 1; j >= 0; j-- )
368  {
369  x = 0.0;
370  Vector< std::complex<double> > ad_v( j + 2 );
371  for ( int jj = 0; jj < j + 2; jj++ ) ad_v[ jj ] = ad[ jj ];
372  laguer( ad_v, x, its );
373  if ( std::abs( std::imag(x) ) <=
374  2.0 * EPS * std::abs( std::real( x ) ) )
375  {
376  x = std::complex<double>( std::real( x ), 0.0 );
377  }
378  poly_roots[ j ] = x;
379  b = ad[ j + 1 ];
380  for ( int jj = j; jj >= 0; jj-- ) {
381  c = ad[ jj ];
382  ad[ jj ] = b ;
383  b = x * b + c;
384  }
385  }

424  template <typename T>
425  inline void Polynomial<T>::laguer( Vector< std::complex<double> >& a,
426  std::complex<double>& x,
427  std::size_t& iterations )
428  {
429  const int MR=8, MT=10, MAXIT=MT*MR;
430  const double EPS=std::numeric_limits<double>::epsilon();
431  // Fraction used to break possible limit cycle
432  static const double frac[MR+1]=
433  {0.0,0.5,0.25,0.75,0.13,0.38,0.62,0.88,1.0};
434  std::complex<double> dx,x1,b,d,f,g,h,sq,gp,gm,g2;
435  int m=a.size()-1;
436  for ( int iter = 1; iter <= MAXIT; iter++ )
437  {
438  iterations=iter;
439  b = a[ m ];
440  double err = std::abs( b );
441  d = f = 0.0;
442  double abx = std::abs( x );
443  for (int j = m - 1; j >= 0 ; j-- )
444  {
445  f = x * f + d;
446  d = x * d + b;
447  b = x * b + a[ j ];
448  err = std::abs( b ) + abx * err;
449  }
450  err *= EPS;
451  if ( std::abs( b ) <= err ){ return; }
452  g = d / b;
453  g2 = g * g;
454  h = g2 - 2.0 * f / b;
455  sq = std::sqrt( double( m - 1 ) * ( double( m ) * h - g2 ) );
456  gp = g + sq;
457  gm = g - sq;
458  double abp = std::abs( gp );
459  double abm = std::abs( gm );
460  if ( abp < abm ){ gp = gm; }
461  dx = std::max( abp, abm ) > 0. ? double( m ) / gp :
462  std::polar( 1 + abx, double( iter ));
463  x1 = x - dx;
464  if ( x == x1 ){ return; }
465  if ( iter % MT != 0 ){ x = x1; }
466  else { x -= frac[ iter / MT ] * dx; }
467  }
468  throw Error( "Polynomial: Too many iterations in laguer method" );
469  }
