1 /// \file ODE_BVP.h
2 /// A class for solving a system of \f$ n \f$ first-order ordinary differential
3 /// equations subject to \f$ n \f$ boundary conditions defined at the ends of
4 /// the domain \f$ x = x_{left} \f$ or \f$ x_{right} \f$. The system is defined
5 /// by \f[ M_0 \frac{d \mathbf{f}}{dx} = \mathbf{R} \f] where
6 /// \f$ \mathbf{f}=\mathbf{f}(x) \f$ is the vector on unknowns,
7 /// \f$ \mathbf{R} = \mathbf{R}(\mathbf{f}, x) \f$ is the vector of residuals
8 /// and \f$ M_0 = M_0(\mathbf{f}, x) \f$ is a matrix.
9 ///
10 /// The system is solved via Newton iteration by splitting the solution into a
11 /// known part \f$ \mathbf{f}^g \f$ and a correction \f$ \mathbf{f}^c \f$ such
12 /// that \f$ \mathbf{f} =\mathbf{f}^g + \mathbf{f}^c \f$. The matrix \f$ M_0 \f$
13 /// and the residual \f$ \mathbf{R} \f$ are exapanded about the known solution
14 /// such that \f[ \mathbf{R}(\mathbf{f}, x) = \mathbf{R}(\mathbf{f}^g, x) +
15 /// \frac{\partial \mathbf{R}}{\partial \mathbf{f}} \bigg\vert_{\mathbf{f}^g}
16 /// \mathbf{f}^c + O((\mathbf{f}^c)^2), \hspace{0.5cm} M_0(\mathbf{f}, x) =
17 /// M_0(\mathbf{f}^g, x) + \frac{\partial M_0}{\partial \mathbf{f}}
18 /// \bigg\vert_{\mathbf{f}^g} \mathbf{f}^c + O((\mathbf{f}^c)^2).\f]
19 /// The system is linearised for the corrections such that \f[ M_0(\mathbf{f}^g,
20 /// x) \frac{d \mathbf{f}^c}{d x} + \left(\frac{\partial M_0}{\partial
21 /// \mathbf{f}} \bigg\vert_{\mathbf{f}^g} - \frac{\partial \mathbf{R}}{\partial
22 /// \mathbf{f}} \bigg\vert_{\mathbf{f}^g} \right) \mathbf{f}^c = \mathbf{R}
23 /// - M_0(\mathbf{f}^g, x) \frac{d \mathbf{f}^g}{d x}. \f] The system is then
24 /// solved for the correction \f$ \mathbf{f}^c \f$ which is used to update the
25 /// known solution \f$ \mathbf{f}^g \f$, this is repeated until convergence is
26 /// achieved.

189  template <typename T, typename X>
190  void ODE_BVP<T, X>::solve_bvp()
191  {
192  unsigned order( ptr_EQUATION -> get_order() );
193  unsigned n( SOLUTION.nnodes() );
194  double max_residual( 1.0 );
195  int counter( 0 );
196  int det_sign( LAST_DET_SIGN );
197  // Banded LHS matrix
198  BandedMatrix<T> a( n * order, 2 * order - 1, 2 * order - 1 );
199  // RHS vector
200  Vector<T> b( n * order, 0.0 );
201  // Loop until convergence
202  do {
203  ++counter;
204  assemble_matrix_problem( a, b );
205  //TODO actions before linear solve ?
206  max_residual = b.norm_inf();
207  b = a.solve( b );
208  // Put the solution into the 1D mesh
209  for ( std::size_t var = 0; var < order; ++var )
210  {
211  for ( std::size_t i = 0; i < n; ++i )
212  {
213  SOLUTION( i, var ) += b[ i * order + var ];
214  }
215  }
216 
217  } while( ( max_residual > TOL ) && ( counter < MAX_ITER ) );
218 
219  if ( counter >= MAX_ITER )
220  {
221  std::string problem( "solve_bvp method error: Too many iterations" );
222  throw Error( problem );
223  }
224 
225  if ( MONITOR_DET )
226  {
227  T det = a.det();
228  if ( std::real( det ) < 0 ) {
229  det_sign = -1;
230  } else {
231  det_sign = 1;
232  }
233  if ( det_sign * LAST_DET_SIGN < 0 )
234  {
235  LAST_DET_SIGN = det_sign;
236  std::string problem;
237  problem = "Determinant monitor has changed signs in ODE_BVP.\n";
238  problem += "Bifurcation detected.\n";
239  throw Error( problem );
240  }
241  LAST_DET_SIGN = det_sign;
242  }

268  template <typename T, typename X>
269  std::pair<unsigned, unsigned> ODE_BVP<T, X>::adapt( const double& adapt_tol )
270  {
271  unsigned order( ptr_EQUATION -> get_order() );
272  unsigned n( SOLUTION.nnodes() );
273 
274  Vector<T> F_node( order, 0.0 );
275  Vector<T> R_node( order, 0.0 );
276 
277  std::vector<bool> refine( n, false );
278  std::vector<bool> unrefine( n, false );
279 
280  Vector<double> Res2( n, 0.0 );
281 
282  for ( std::size_t node = 1; node <= n - 2; node += 2 )
283  {
284  // set the current solution at this node
285  for ( unsigned var = 0; var < order; ++var )
286  {
287  F_node[ var ] = SOLUTION( node, var );
288  }
289  // set the y-pos in the eqn
290  ptr_EQUATION -> coord(0) = SOLUTION.coord( node );
291  // Update the equation to the nodal position
292  ptr_EQUATION -> update( F_node );
293  // step size centred at the node
294  X invh = 1. / ( SOLUTION.coord( node + 1 ) - SOLUTION.coord( node - 1 ) );
295  // loop over all the variables
296  Vector<T> temp( order, 0.0 );
297  for ( unsigned var = 0; var < order; ++var )
298  {
299  temp[ var ] = ptr_EQUATION -> residual()[ var ] - (
300  SOLUTION( node + 1, var ) - SOLUTION( node - 1, var ) ) * invh;
301  }
302  Res2[ node ] = temp.norm_inf();
303  if ( Res2[ node ] > adapt_tol )
304  {
305  refine[ node ] = true;
306  }
307  else if ( Res2[ node ] < TOL / 10. )
308  {
309  unrefine[ node ] = true;
310  }
311  }
312 
313  std::size_t no_refined( 0 ), no_unrefined( 0 );
314  for ( std::size_t i = 0; i < refine.size(); ++i )
315  {
316  if ( refine[ i ] == true ){ no_refined++; }
317  if ( unrefine[ i ] == true ){ no_unrefined++; }
318  }
319 
320  // make a new refined/unrefined mesh
321  Vector<X> nodes( SOLUTION.nodes() );
322  Vector<X> newX;
323  newX.push_back( nodes[ 0 ] );
324  for ( std::size_t i = 1; i < n - 1; ++i )
325  {
326  if ( refine[ i ] )
327  {
328  if ( !refine[ i - 1 ] )
329  {
330  // Have not already refined to the left
331  // so refine left AND right with new nodes
332  X left( nodes[ i - 1 ] );
333  X right( nodes[ i + 1 ] );
334  X here( nodes[ i ] );
335  newX.push_back( ( left + here ) / 2. );
336  newX.push_back( here );
337  newX.push_back( ( right + here ) / 2. );
338  }
339  else
340  {
341  // already left refined, so just refine right
342  X right( nodes[ i + 1 ] );
343  X here( nodes[ i ] );
344  newX.push_back( here );
345  newX.push_back( ( right + here ) / 2. );
346  }
347  }
348  else if ( !unrefine[ i ] )
349  {
350  newX.push_back( nodes[ i ] );
351  }
352  }
353  newX.push_back( nodes[ n - 1 ] );
354 
355  SOLUTION.remesh( newX );
356 
357  LAST_DET_SIGN = 0;
358 
359  std::pair< unsigned, unsigned > feedback;
360  feedback.first = no_refined;
361  feedback.second = no_unrefined;
362  return feedback;
363  }

405  template <typename T, typename X>
406  double ODE_BVP<T, X>::arclength_solve( const double& step )
407  {
408  this -> ds() = step;
409  std::size_t order( ptr_EQUATION -> get_order() ); // Order of the equation
410  std::size_t n( SOLUTION.nnodes() ); // Number of nodes
411  std::size_t Size( order * n );
412 
413  BandedMatrix<T> Jac( n * order, 2 * order - 1, 2 * order - 1 ); // Jacobian matrix
414  // Residuals over all nodes
415  Vector<T> Res1( Size, 0.0 );
416  Vector<T> Res2( Size, 0.0 );
417  Vector<T> dRes_dp( Size, 0.0 );
418  // RHS vectors for the linear solvers
419  Vector<T> y( Size, 0.0 );
420  Vector<T> z( Size, 0.0 );
421  Vector<T> Jac_E;
422  // Make backups in case we can't find a converged solution
423  Vector<T> backup_state( SOLUTION.vars_as_vector() );
424  T backup_parameter( *( this -> ptr_PARAM ) );
425  // Generate a 1st order guess for the next state and parameter
426  Vector<T> x( this -> LAST_X + this -> X_DERIV_S * this -> DS );
427  *( this -> ptr_PARAM ) = this -> LAST_PARAM + this -> PARAM_DERIV_S * this -> DS;
428 
429  SOLUTION.set_vars_from_vector( x ); // Update the solution mesh
430  bool step_succeeded( false ); // Check for success
431  std::size_t itn( 0 ); // Iteration counter
432 
433  do
434  {
435  ++itn;
436  double E1 = this -> arclength_residual( x ); // Arclength residual
437  assemble_matrix_problem( Jac, Res1 ); // Assemble matrix problem
438  if ( Res1.norm_inf() < TOL && itn > 1 )
439  {
440  step_succeeded = true;
441  break;
442  }
443  y = Jac.solve( Res1 );
444  // Derivatives wrt parameter
445  const double delta( 1.e-8 );
446  *( this -> ptr_PARAM ) += delta;
447  assemble_matrix_problem( Jac, Res2 );
448  double E2 = this -> arclength_residual( x );
449  *( this -> ptr_PARAM ) -= delta;
450  dRes_dp = ( Res2 - Res1 ) / delta;
451  double dE_dp = ( E2 - E1 ) / delta;
452  z = Jac.solve( dRes_dp );
453  Jac_E = this -> Jac_arclength_residual( x );
454  T delta_p = -( E1 + Jac_E.dot( y ) ) / ( dE_dp + Jac_E.dot( z ) );
455  Vector<T> delta_x = y + z * delta_p;
456  // Update the state variables and the parameter with the corrections
457  x += delta_x;
458  *( this -> ptr_PARAM ) += delta_p;
459  SOLUTION.set_vars_from_vector( x );
460  // Check for convergence
461  if ( delta_x.norm_inf() < TOL )
462  {
463  step_succeeded = true;
464  break;
465  }
466  if ( itn > MAX_ITER )
467  {
468  step_succeeded = false;
469  break;
470  }
471  }while( true );
472  // If this isn't a successful step
473  if ( !step_succeeded )
474  {
475  // Restore things using the backups we made earlier
476  SOLUTION.set_vars_from_vector( backup_state ); // Restore state
477  *( this -> ptr_PARAM ) = backup_parameter; // Restore parameter
478  this -> DS /= this -> ARCSTEP_MULTIPLIER; // Reduce step length
479  }
480  else
481  {
482  this -> update( SOLUTION.vars_as_vector() );
483  if ( itn > 8 || std::abs( this -> DS ) > this -> MAX_DS )
484  {
485  // Converging too slowly, so decrease DS
486  this -> DS /= this -> ARCSTEP_MULTIPLIER;
487  }
488  if ( itn < 4 ) // Converging too quickly, so increase DS
489  {
490  if ( std::abs( this -> DS * this -> ARCSTEP_MULTIPLIER ) < this -> MAX_DS )
491  {
492  this -> DS *= this -> ARCSTEP_MULTIPLIER;
493  }
494  }
495  }
496  return this -> DS;
497  }

501  template <typename T, typename X>
502  void ODE_BVP<T, X>::assemble_matrix_problem( BandedMatrix<T>& a,
503  Vector<T>& b )
504  {
505  a.fill( 0.0 );
506  unsigned order( ptr_EQUATION -> get_order() );
507  unsigned n( SOLUTION.nnodes() );
508  std::size_t row( 0 );
509 
510  Matrix<T> Jac_midpt( order, order, 0.0 );
511  Vector<T> F_midpt( order, 0.0 );
512  Vector<T> R_midpt( order, 0.0 );
513  Vector<T> state_dy( order, 0.0 );
514  Matrix<T> h0( order, order, 0.0 );
515  // update the BC residuals
516  ptr_LEFT_RESIDUAL -> update( SOLUTION.get_nodes_vars( 0 ) );
517  // add the LHS BCs to the matrix problem
518  for ( std::size_t i = 0; i < ptr_LEFT_RESIDUAL -> get_order(); ++i )
519  {
520  // loop thru variables at LHS of the domain
521  for ( std::size_t var = 0; var < order; ++var )
522  {
523  a( row, var ) = ptr_LEFT_RESIDUAL -> jacobian()( i, var );
524  }
525  b[ row ] = - ptr_LEFT_RESIDUAL -> residual()[ i ];
526  ++row;
527  }
528  // inner nodes
529  for ( std::size_t node = 0; node <= n - 2; ++node )
530  {
531  const std::size_t lnode( node );
532  const std::size_t rnode( node + 1 );
533  // inverse of step size
534  const X invh = 1. / ( SOLUTION.coord( rnode ) - SOLUTION.coord( lnode ) );
535  for ( std::size_t var = 0; var < order; ++var )
536  {
537  F_midpt[ var ] = ( SOLUTION( lnode, var ) +
538  SOLUTION( rnode, var ) ) / 2.;
539  state_dy[ var ] = ( SOLUTION( rnode, var ) -
540  SOLUTION( lnode, var ) ) * invh;
541  }
542  X y_midpt = ( SOLUTION.coord( lnode ) + SOLUTION.coord( rnode ) ) / 2.;
543  ptr_EQUATION -> coord( 0 ) = y_midpt;
544  ptr_EQUATION -> update( F_midpt );
545  ptr_EQUATION -> get_jacobian_of_matrix0_mult_vector( F_midpt,
546  state_dy, h0 );
547  for ( std::size_t var = 0; var < order; ++var )
548  {
549  for ( std::size_t i = 0; i < order; ++i )
550  {
551  // Jac of matrix * F_y * g terms
552  int col_l( order * lnode + i );
553  int col_r( order * rnode + i );
554  a( row, col_l ) += h0( var, i )/2.;
555  a( row, col_r ) += h0( var, i )/2.;
556  // Matrix * g_y terms
557  a( row, col_l ) -= ptr_EQUATION -> matrix0()( var, i ) * invh;
558  a( row, col_r ) += ptr_EQUATION -> matrix0()( var, i ) * invh;
559  // Jacobian of RHS terms
560  a( row, col_l ) -= ptr_EQUATION -> jacobian()( var, i ) / 2.;
561  a( row, col_r ) -= ptr_EQUATION -> jacobian()( var, i ) / 2.;
562  }
563  b[ row ] = ptr_EQUATION -> residual()[ var ]
564  - state_dy.dot( ptr_EQUATION -> matrix0()[ var ] );
565  row++;
566  }
567  }
568  // update the BC residuals
569  ptr_RIGHT_RESIDUAL -> update( SOLUTION.get_nodes_vars( n - 1 ) );
570  // add the RHS BCs to the matrix problem
571  for ( std::size_t i = 0; i < ptr_RIGHT_RESIDUAL -> get_order(); ++i )
572  {
573  // loop thru variables at RHS of the domain
574  for ( std::size_t var = 0; var < order; ++var )
575  {
576  int col( order * ( n - 1 ) + var );
577  a( row, col ) = ptr_RIGHT_RESIDUAL -> jacobian()( i, var );
578  }
579  b[ row ] = - ptr_RIGHT_RESIDUAL -> residual()[ i ];
580  ++row;
581  }
582 
583  if ( row != n * order )
584  {
585  std::string problem;
586  problem += "ODE_BVP error: incorrect number of boundary conditions.";
587  throw Error( problem );
588  }
589  }
