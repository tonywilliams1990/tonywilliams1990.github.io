21  /// \param z The argument of the function (not a negative integer or zero)
22  /// \return The gamma function \f$\Gamma(z)\f$ using Spouge's approximation
23  template <typename T>
24  T gamma( const T& z )
25  {
26  const int a = 12;
27  Vector<T> c;
28  c.resize( a );
29  int k;
30  T accm;
31 
32  double factrl = 1.0; // (k - 1)!*(-1)^k with 0!==1
33  c[ 0 ] = std::sqrt( 2.0 * M_PI );
34  for(k=1; k < a; k++) {
35  c[ k ] = std::exp( 1. * a - k )
36  * std::pow( 1. * a - k, k - 0.5 ) / factrl;
37  factrl *= -k;
38  }
39  accm = c[ 0 ];
40  for( k = 1; k < a; k++ )
41  {
42  accm += c[ k ] / ( z + 1. * k );
43  }
44  accm *= std::exp( - ( z + 1. * a ) ) * std::pow( z + 1. * a, z + 0.5 );
45  return accm / z; // Gamma(z) = Gamma(z+1) / z
46  }
