132  J = this->ptr_RESIDUAL -> jacobian(); // Get the Jacobian
133  R1 = this->ptr_RESIDUAL -> residual(); // Get the residual
134  double E1 = this -> arclength_residual( x ); // Arclength residual
135  // Compute derivatives wrt the parameter
136  *(this->ptr_PARAM) += this->DELTA;
137  this->ptr_RESIDUAL -> residual_fn( x, R2 );
138  double E2 = this -> arclength_residual( x );
139  *(this->ptr_PARAM) -= this->DELTA;
140  dR_dp = ( R2 - R1 ) / this->DELTA;
141  T dE_dp = ( E2 - E1 ) / this->DELTA;
142  // Bordering algorithm
143  y = J.solve( -R1 );
144  z = J.solve( -dR_dp );
145  JacE = this->Jac_arclength_residual( x );
146  T delta_p = -( E1 + JacE.dot( y ) ) / ( dE_dp + JacE.dot( z ) );
147  Vector<T> delta_x = y + z * delta_p;
148  double max_correction = std::max( delta_x.norm_inf(),
149  std::abs( delta_p ) );
