4 /// The Falkner-Skan equation \f[ f'''(\eta) + f(\eta)f''(\eta) + \beta \left[1
5 /// - \left(f'(\eta) \right)^2 \right] = 0 \f] is solved on \f$ \eta \in
6 /// [0,\infty) \f$ subject to the boundary conditions \f$ f(0) = f'(0) = 0\f$
7 /// and \f$ f'(\eta) \to 1 \f$ as \f$ \eta \to \infty\f$.
8 /// The system is solved using rational Chebyshev functions on a semi-infinite
9 /// interval \f$ TL_i(\eta; L) \f$. The rational chebyshev functions all have zero
10 /// derivative as \f$ \eta \to \infty \f$ so it is necessary to use the change
11 /// of variables \f$ u(\eta) = f(\eta) - \eta \f$. The Falkner-Skan system is
12 /// transformed to \f[ u''' + (\eta + u)u'' - \beta u'\left(2 + u' \right) = 0
13 /// \f] subject to the boundary conditions \f$ u(0) = 0\f$, \f$ u'(0) = -1 \f$
14 /// and \f$ u' \to 0 \f$ as \f$ \eta \to \infty \f$. This nonlinear equation is
15 /// solver using Newton iteration on the spectral coefficients.

102  F[ i ] += beta * u_g( yi, 1 ) * ( 2 + u_g( yi, 1 ) );
103  }
104

106  for ( std::size_t j = 0; j < n + 2; j++ )
107  {

109  M( n, j ) = rationalsemi( yi, j );
110  }
111  F[ n ] = - u_g( 0.0 );
112

115  {
116  double yi = 0.0;

118  }

120

140  solution( i, 2 ) = u_g( y, 2 ); // f'' = u''
